Mapa do Gestor — Situação Sanitária das Escolas

/* ================= MAPA ================= */ const map = L.map("map").setView([-3.7319, -38.5267], 12); L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", { attribution: "© OpenStreetMap" }).addTo(map); /* ================= CORES ================= */ function corStatus(status) { if (status === "Condição crítica") return "#dc2626"; if (status === "Atenção elevada") return "#f59e0b"; if (status === "Situação de alerta") return "#facc15"; return "#16a34a"; } /* ================= LAYERS ================= */ const escolasLayer = L.layerGroup().addTo(map); let bairrosLayer = null; /* ================= CARREGAR AVALIAÇÕES ================= */ db.collection("avaliacoes").onSnapshot(snapshot => { escolasLayer.clearLayers(); const porEscola = {}; const porBairro = {}; snapshot.forEach(doc => { const d = doc.data(); if (!d.escola) return; // Mantém a avaliação mais recente por escola if (!porEscola[d.escola] || doc.createTime > porEscola[d.escola].time) { porEscola[d.escola] = { ...d, time: doc.createTime }; } }); Object.values(porEscola).forEach(d => { const base = window.escolas.find(e => e.nome === d.escola); if (!base) return; // Contagem por bairro const bairro = base.bairro || "Sem bairro"; if (!porBairro[bairro]) { porBairro[bairro] = { total: 0, critico: 0 }; } porBairro[bairro].total++; if (d.classe === "critico") porBairro[bairro].critico++; const marker = L.circleMarker( [base.lat, base.lng], { radius: 7, color: corStatus(d.status), fillColor: corStatus(d.status), fillOpacity: 0.9, className: "pulse" } ); marker.bindPopup(` ${d.escola}
Status: ${d.status}
Pontuação: ${d.pontuacao} `); escolasLayer.addLayer(marker); }); carregarBairros(porBairro); }); /* ================= BAIRROS ================= */ function carregarBairros(dadosBairro) { if (bairrosLayer) bairrosLayer.remove(); fetch("bairros.geojson") .then(r => r.json()) .then(geojson => { bairrosLayer = L.geoJSON(geojson, { style: f => { const nome = f.properties.nome; const d = dadosBairro[nome]; let pct = 0; if (d) pct = (d.critico / d.total) * 100; let cor = "#d1fae5"; if (pct >= 50) cor = "#fee2e2"; else if (pct >= 20) cor = "#ffedd5"; else if (pct >= 5) cor = "#fef9c3"; return { weight: 1, color: "#374151", fillColor: cor, fillOpacity: 0.6 }; }, onEachFeature: (f, layer) => { const nome = f.properties.nome; const d = dadosBairro[nome]; let texto = "Sem escolas avaliadas"; if (d) { texto = ` Escolas: ${d.total}
Críticas: ${d.critico}
% crítico: ${((d.critico/d.total)*100).toFixed(1)}% `; } layer.bindPopup(`${nome}
${texto}`); } }).addTo(map); }); }
